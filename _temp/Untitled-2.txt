# Load the xml data from a file or a string
#[xml]$xml = Get-Content "config.xml"
$smlString = "<?xml version='1.0' encoding='utf-8' ?>
<configuration>
  <app title='Simple Configuration UI'>
    <text:database-provider default='SQLClient' />
    <number:rety-max default='10' />
    <boolean:enable-mfa default='true' />
    <sselect:binding default='http-artifact'>
      <item:http-binding value='http' />
      <item:http-artifact value='artifact' />
    </sselect:binding>
    <mselect:attributes default='fqn;email'>
      <item:fqn value='Fully Qualified Name' />
      <item:email value='Email' />
      <item:role value='Role' />
    </mselect:attributes>
  </app>
  <data>
    <item:database-provider value='PostgresClient' />
  </data>
</configuration>"
[xml]$xml = $xmlString

# Get the app element and its title attribute
$app = $xml.configuration.app
$title = $app.title

# Get the data element and its child elements
$data = $xml.configuration.data
$dataItems = $data.ChildNodes

# Create a new WPF window with the title
$window = New-Object System.Windows.Window
$window.Title = $title

# Create a grid to hold the components
$grid = New-Object System.Windows.Controls.Grid
$grid.Margin = New-Object System.Windows.Thickness 10

# Define the rows and columns of the grid
$grid.RowDefinitions.Add((New-Object System.Windows.Controls.RowDefinition))
$grid.RowDefinitions.Add((New-Object System.Windows.Controls.RowDefinition))
$grid.RowDefinitions.Add((New-Object System.Windows.Controls.RowDefinition))
$grid.RowDefinitions.Add((New-Object System.Windows.Controls.RowDefinition))
$grid.RowDefinitions.Add((New-Object System.Windows.Controls.RowDefinition))
$grid.ColumnDefinitions.Add((New-Object System.Windows.Controls.ColumnDefinition))
$grid.ColumnDefinitions.Add((New-Object System.Windows.Controls.ColumnDefinition))

# Loop through the child elements of the app element
$row = 0
foreach ($element in $app.ChildNodes)
{
    # Get the name and value of the element
    $name = $element.LocalName
    $value = $element.default

    # Create a label for the element name
    $label = New-Object System.Windows.Controls.Label
    $label.Content = ToTitleCase($name.Split(':')[1])
    $label.HorizontalAlignment = [System.Windows.HorizontalAlignment]::Right
    $label.VerticalAlignment = [System.Windows.VerticalAlignment]::Center
    $label.Margin = New-Object System.Windows.Thickness 5

    # Add the label to the grid at the current row and column 0
    [System.Windows.Controls.Grid]::SetRow($label, $row)
    [System.Windows.Controls.Grid]::SetColumn($label, 0)
    $grid.Children.Add($label)

    # Check the type of the element by its prefix
    switch ($name.Split(':')[0])
    {
        "text" {
            # Create a textbox for the text element
            $textBox = New-Object System.Windows.Controls.TextBox
            $textBox.Text = $value
            $textBox.HorizontalAlignment = [System.Windows.HorizontalAlignment]::Left
            $textBox.VerticalAlignment = [System.Windows.VerticalAlignment]::Center
            $textBox.Margin = New-Object System.Windows.Thickness 5

            # Add a handler for the text changed event of the textbox
            Register-ObjectEvent -InputObject $textBox -EventName TextChanged -Action {
                # Get the identifier of the text element
                $id = $name.Split(':')[1]

                # Find the corresponding data item element by its name
                $dataItem = ($dataItems | Where-Object { $_.LocalName -eq "item:$id" })

                # If the data item element exists, update its value attribute with the textbox text
                if ($dataItem)
                {
                    $dataItem.value = $textBox.Text
                }
            }

            # Add the textbox to the grid at the current row and column 1
            [System.Windows.Controls.Grid]::SetRow($textBox, $row)
            [System.Windows.Controls.Grid]::SetColumn($textBox, 1)
            $grid.Children.Add($textBox)
        }

        "number" {
            # Create a numeric updown for the number element
            Add-Type -AssemblyName WPFToolkit.Extended # This assembly contains a NumericUpDown control for WPF
            $numericUpDown = New-Object Xceed.Wpf.Toolkit.NumericUpDown 
            $numericUpDown.Value = [int]$value
            $numericUpDown.HorizontalAlignment = [System.Windows.HorizontalAlignment]::Left
            $numericUpDown.VerticalAlignment = [System.Windows.VerticalAlignment]::Center
            $numericUpDown.Margin = New-Object System.Windows.Thickness 5

            # Add a handler for the value changed event of the numeric updown
            Register-ObjectEvent -InputObject $numericUpDown -EventName ValueChanged -Action {
                # Get the identifier of the number element
                $id = $name.Split(':')[1]

                # Find the corresponding data item element by its name
                $dataItem = ($dataItems | Where-Object { $_.LocalName -eq "item:$id" })

                # If the data item element exists, update its value attribute with the numeric updown value
                if ($dataItem)
                {
                    $dataItem.value = $numericUpDown.Value.ToString()
                }
            }

            # Add the numeric updown to the grid at the current row and column 1
            [System.Windows.Controls.Grid]::SetRow($numericUpDown, $row)
            [System.Windows.Controls.Grid]::SetColumn($numericUpDown, 1)
            $grid.Children.Add($numericUpDown)
        }

        "boolean" {
            # Create a checkbox for the boolean element
            $checkBox = New-Object System.Windows.Controls.CheckBox
            $checkBox.IsChecked = [bool]$value
            $checkBox.HorizontalAlignment = [System.Windows.HorizontalAlignment]::Left
            $checkBox.VerticalAlignment = [System.Windows.VerticalAlignment]::Center
            $checkBox.Margin = New-Object System.Windows.Thickness 5

            # Add a handler for the checked and unchecked events of the checkbox
            Register-ObjectEvent -InputObject $checkBox -EventName Checked -Action {
                # Get the identifier of the boolean element
                $id = $name.Split(':')[1]

                # Find the corresponding data item element by its name
                $dataItem = ($dataItems | Where-Object { $_.LocalName -eq "item:$id" })

                # If the data item element exists, update its value attribute with true
                if ($dataItem)
                {
                    $dataItem.value = "true"
                }
            }

            Register-ObjectEvent -InputObject $checkBox -EventName Unchecked -Action {
                # Get the identifier of the boolean element
                $id = $name.Split(':')[1]

                # Find the corresponding data item element by its name
                $dataItem = ($dataItems | Where-Object { $_.LocalName -eq "item:$id" })

                # If the data item element exists, update its value attribute with false
                if ($dataItem)
                {
                    $dataItem.value = "false"
                }
            }

            # Add the checkbox to the grid at the current row and column 1
            [System.Windows.Controls.Grid]::SetRow($checkBox, $row)
            [System.Windows.Controls.Grid]::SetColumn($checkBox, 1)
            $grid.Children.Add($checkBox)
        }

        "sselect" {
            # Create a combobox for the single select element
            $comboBox = New-Object System.Windows.Controls.ComboBox
            $comboBox.SelectedValuePath = "value"
            $comboBox.DisplayMemberPath = "label"
            $comboBox.SelectedValue = $value
            $comboBox.HorizontalAlignment = [System.Windows.HorizontalAlignment]::Left
            $comboBox.VerticalAlignment = [System.Windows.VerticalAlignment]::Center
            $comboBox.Margin = New-Object System.Windows.Thickness 5
            # Loop through the item elements of the single select element
            foreach ($item in $element.ChildNodes)
            {
                # Get the name and value of the item element
                $itemName = $item.LocalName
                $itemValue = $item.value

                # Create an object to represent the item with its value and label properties
                $itemObject = New-Object PSObject -Property @{value = $itemValue; label = ToTitleCase($itemName.Split(':')[1])}

                # Add the item object to the combobox items collection
                $comboBox.Items.Add($itemObject)
            }

            # Add a handler for the selection changed event of the combobox
            Register-ObjectEvent -InputObject $comboBox -EventName SelectionChanged -Action {
                # Get the identifier of the single select element
                $id = $name.Split(':')[1]

                # Find the corresponding data item element by its name
                $dataItem = ($dataItems | Where-Object { $_.LocalName -eq "item:$id" })

                # If the data item element exists, update its value attribute with the combobox selected value
                if ($dataItem)
                {
                    $dataItem.value = $comboBox.SelectedValue.ToString()
                }
            }

            # Add the combobox to the grid at the current row and column 1
            [System.Windows.Controls.Grid]::SetRow($comboBox, $row)
            [System.Windows.Controls.Grid]::SetColumn($comboBox, 1)
            $grid.Children.Add($comboBox)
        }

        "mselect" {
            # Create a listbox for the multi select element
            $listBox = New-Object System.Windows.Controls.ListBox
            $listBox.SelectedValuePath = "value"
            $listBox.DisplayMemberPath = "label"
            $listBox.SelectionMode = [System.Windows.Controls.SelectionMode]::Multiple
            $listBox.HorizontalAlignment = [System.Windows.HorizontalAlignment]::Left
            $listBox.VerticalAlignment = [System.Windows.VerticalAlignment]::Center
            $listBox.Margin = New-Object System.Windows.Thickness 5

            # Loop through the item elements of the multi select element
            foreach ($item in $element.ChildNodes)
            {
                # Get the name and value of the item element
                $itemName = $item.LocalName
                $itemValue = $item.value

                # Create an object to represent the item with its value and label properties
                $itemObject = New-Object PSObject -Property @{value = $itemValue; label = ToTitleCase($itemName.Split(':')[1])}

                # Add the item object to the listbox items collection
                $listBox.Items.Add($itemObject)

                # Check if the item value is in the default value of the multi select element
                if ($value.Contains($itemValue))
                {
                    # Select the item in the listbox
                    $listBox.SelectedItems.Add($itemObject)
                }
            }

            # Add a handler for the selection changed event of the listbox
            Register-ObjectEvent -InputObject $listBox -EventName SelectionChanged -Action {
                # Get the identifier of the multi select element
                $id = $name.Split(':')[1]

                # Find the corresponding data item element by its name
                $dataItem = ($dataItems | Where-Object { $_.LocalName -eq "item:$id" })

                # If the data item element exists, update its value attribute with a semicolon-separated string of listbox selected values
                if ($dataItem)
                {
                    $selectedValues = ($listBox.SelectedItems | ForEach-Object { $_.value }) -join ";"
                    $dataItem.value = $selectedValues
                }
            }

            # Add the listbox to the grid at the current row and column 1
            [System.Windows.Controls.Grid]::SetRow($listBox, $row)
            [System.Windows.Controls.Grid]::SetColumn($listBox, 1)
            $grid.Children.Add($listBox)
        }
    }

    # Increment the row index
    $row++
}

# Set the content of the window to the grid
$window.Content = $grid

# Show the window
$window.ShowDialog()
