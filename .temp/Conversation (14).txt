using Sitecore.Mvc.Presentation;

namespace SitecoreDemo

    public class ParameterRenderingModel : RenderingModel

        // Single-line text field
        public string Text { get; set; }

        public override void Initialize(Rendering rendering)
        {
            base.Initialize(rendering);

            // Get the rendering parameters item
            var parameters = rendering.Parameters;

            // Get the RTE field value
            RichText = parameters["RTE"];

            // Get the multi-list field value
            var selectedIds = parameters["Multi-list"].Split('|');

            // Get the selected items from the database
            var database = Sitecore.Context.Database;
            SelectedItems = selectedIds.Select(id => database.GetItem(id)).ToArray();

            // Get the single-line text field value
            Text = parameters["Single-Line Text"];
        }
    }
}

[csharp]
using Sitecore.Data.Items;
using Sitecore.Mvc.Presentation;

namespace SitecoreDemo
{
    public class ParameterRenderingModel : RenderingModel
    {
        // RTE field
        public string RichText { get; set; }

        // Multi-list field
        public Item[] SelectedItems { get; set; }

        // Single-line text field
        public string Text { get; set; }

        // Database service
        private readonly IDatabaseService _databaseService;

        // Constructor with dependency injection
        public ParameterRenderingModel(IDatabaseService databaseService)
        {
            _databaseService = databaseService;
        }

        public override void Initialize(Rendering rendering)
        {
            base.Initialize(rendering);

            // Get the rendering parameters item
            var parameters = rendering.Parameters;

            // Get the RTE field value
            RichText = parameters["RTE"];

            // Get the multi-list field value
            var selectedIds = parameters["Multi-list"].Split('|');

            // Get the selected items from the database service
            SelectedItems = selectedIds.Select(id => _databaseService.GetItem(id)).ToArray();

            // Get the single-line text field value
            Text = parameters["Single-Line Text"];
        }
    }
}

can you explain more about RenderingModel?
## Bing
Sure, I can explain more about RenderingModel. RenderingModel is a class that implements the IRenderingModel interface and provides access to the context item, the datasource item, and the rendering parameters item. It is the default model for view renderings, which are the simplest type of renderings in Sitecore MVC. You can use RenderingModel to output simple data from content items using Sitecore's field helper. You can also create your own custom models that inherit from RenderingModel or IRenderingModel and override the Initialize method to add more properties or logic.

## Bing
Sure, I can show you an example of using Sitecore's field helper. Sitecore's field helper is a set of components that allow you to output field values in a Sitecore MVC view. For example, if you want to output a text field, you can use the @Html.Sitecore().Field() method and pass the field name and optionally the item as parameters. Here is an example:

[html]
@using Sitecore.Mvc
@using Sitecore.Mvc.Presentation
@model RenderingModel

<h1>@Html.Sitecore().Field("Title", Model.Item)</h1>
This will output the value of the Title field of the current item or the datasource item if specified. You can also use other field helper components for different types of fields, such as images, links, rich text, etc. You can find more examples and documentation on Sitecore's official website .
