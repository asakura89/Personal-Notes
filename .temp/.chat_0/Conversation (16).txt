Here is a possible code snippet for a simple task scheduler in .net 6 using threading and can run action from dll dynamically like plugin. I'm not sure if it works as intended, so please use it with caution and test it thoroughly.

[csharp]
using System;
using System.Collections.Generic;
using System.IO;
using System.Reflection;
using System.Threading;
using System.Threading.Tasks;

namespace TaskScheduler
{
    class Program
    {
        // A list of tasks to run
        static List<TaskInfo> tasks = new List<TaskInfo>();

        // A cancellation token source to stop the scheduler
        static CancellationTokenSource cts = new CancellationTokenSource();

        static void Main(string[] args)
        {
            // Load the tasks from a configuration file
            LoadTasks("tasks.txt");

            // Start the scheduler loop
            SchedulerLoop(cts.Token).Wait();

            // Wait for any key to exit
            Console.WriteLine("Press any key to exit...");
            Console.ReadKey();
        }

        // A method to load the tasks from a configuration file
        static void LoadTasks(string fileName)
        {
            // Read each line from the file
            foreach (var line in File.ReadLines(fileName))
            {
                // Split the line by comma
                var parts = line.Split(',');

                // Parse the parts into a task info object
                var task = new TaskInfo
                {
                    Name = parts[0], // The name of the task
                    Interval = TimeSpan.Parse(parts[1]), // The interval to run the task
                    DllPath = parts[2], // The path of the dll that contains the action
                    TypeName = parts[3], // The name of the type that implements the action interface
                    MethodName = parts[4] // The name of the method that is the action entry point
                };

                // Add the task to the list
                tasks.Add(task);
            }
        }

        // A method to run the scheduler loop
        static async Task SchedulerLoop(CancellationToken cancellationToken)
        {
            // Loop until cancelled
            while (!cancellationToken.IsCancellationRequested)
            {
                // Get the current time
                var now = DateTime.Now;

                // Loop through each task in the list
                foreach (var task in tasks)
                {
                    // Check if the task is due to run
                    if (now - task.LastRun >= task.Interval)
                    {
                        // Run the task asynchronously and ignore any exception
                        try
                        {
                            await RunTaskAsync(task, cancellationToken);
                        }
                        catch (Exception ex)
                        {
                            Console.WriteLine($"Task {task.Name} failed: {ex.Message}");
                        }
                    }
                }

                // Wait for a short time before checking again
                await Task.Delay(100, cancellationToken);
            }
        }

        // A method to run a task asynchronously
        static async Task RunTaskAsync(TaskInfo task, CancellationToken cancellationToken)
        {
            // Update the last run time of the task
            task.LastRun = DateTime.Now;

            // Load the dll assembly that contains the action
            var assembly = Assembly.LoadFrom(task.DllPath);

            // Get the type that implements the action interface
            var type = assembly.GetType(task.TypeName);

            // Create an instance of the type
            var instance = Activator.CreateInstance(type);

            // Get the method that is the action entry point
            var method = type.GetMethod(task.MethodName);

            // Invoke the method with no parameters and cast the result as a Task object
            var actionTask = (Task)method.Invoke(instance, null);

            // Wait for the action task to complete or cancelled
            await actionTask.WithCancellation(cancellationToken);

            // Write a message to indicate the task is done
            Console.WriteLine($"Task {task.Name} done.");
        }
    }

    // A class to store the information of a task
    class TaskInfo
    {
        public string Name { get; set; } // The name of the task
        public TimeSpan Interval { get; set; } // The interval to run the task
        public string DllPath { get; set; } // The path of the dll that contains the action
        public string TypeName { get; set; } // The name of the type that implements the action interface
        public string MethodName { get; set; } // The name of the method that is the action entry point

        public DateTime LastRun { get; set; } = DateTime.MinValue; // The last time the task was run

    }

    public static class TaskExtensions 
    {
      public static async Task WithCancellation(this Task task, CancellationToken cancellationToken) 
      {
          var tcs = new TaskCompletionSource<bool>();
          using(cancellationToken.Register(s => ((TaskCompletionSource<bool>)s).TrySetResult(true), tcs))
              if (task != await Task.WhenAny(task, tcs.Task))
                  throw new OperationCanceledException(cancellationToken);
          await task;
      }
    }
}

How do I create a configuration file for the tasks?
## Bing
You can create a configuration file for the tasks by using a text editor and saving it as tasks.txt in the same folder as the program. The file should have one line for each task, and each line should have five parts separated by commas: the name of the task, the interval to run the task, the path of the dll that contains the action, the name of the type that implements the action interface, and the name of the method that is the action entry point. For example, a line like this:
HelloWorldTask,00:00:10,HelloWorld.dll,HelloWorld.HelloWorldAction,SayHello
means that there is a task named HelloWorldTask that runs every 10 seconds, and it invokes the SayHello method of the HelloWorldAction type in the HelloWorld.dll assembly.
