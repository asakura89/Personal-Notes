using System.Net;
using System.Net.Mail;

// Create a new mail message object
MailMessage message = new MailMessage();

message.From = new MailAddress("sender@example.com");

message.Subject = "Hello from Bing";
message.Body = "This is a test email generated by Bing";

// Create a new SMTP client object
SmtpClient client = new SmtpClient();

// Set the host, port, credentials and enable SSL for the client
client.Host = "smtp.example.com";
client.Port = 587;
client.Credentials = new NetworkCredential("username", "password");
client.EnableSsl = true;

// Send the email using the client
client.Send(message);

To convert the code to a reusable code inside a .dll library, you can create a new class library project in Visual Studio and add a reference to System.Net.Mail. Then, you can create a public class that encapsulates the email functionality, such as:

using System.Net;
using System.Net.Mail;

namespace EmailLibrary
{
    public class EmailSender
    {
        // Declare private fields for the SMTP client and the mail message
        private SmtpClient _client;
        private MailMessage _message;

        // Create a constructor that takes the sender, recipient, subject and body as parameters
        public EmailSender(string sender, string recipient, string subject, string body)
        {
            // Initialize the mail message object
            _message = new MailMessage();

            // Set the sender, recipient, subject and body of the email
            _message.From = new MailAddress(sender);
            _message.To.Add(new MailAddress(recipient));
            _message.Subject = subject;
            _message.Body = body;

            // Initialize the SMTP client object
            _client = new SmtpClient();

            // Set the host, port, credentials and enable SSL for the client
            _client.Host = "smtp.example.com";
            _client.Port = 587;
            _client.Credentials = new NetworkCredential("username", "password");
            _client.EnableSsl = true;
        }

        // Create a public method to add CC recipients
        public void AddCC(string cc)
        {
            _message.CC.Add(new MailAddress(cc));
        }

        // Create a public method to add BCC recipients
        public void AddBCC(string bcc)
        {
            _message.Bcc.Add(new MailAddress(bcc));
        }

        // Create a public method to send the email using the client
        public void Send()
        {
            _client.Send(_message);
        }
    }
}

How do I send an email asynchronously?
## Bing
To send an email asynchronously, you can use the SendAsync method of the SmtpClient class, such as:

// Send the email asynchronously using the client
client.SendAsync(message, null);

You can also pass an object as the second parameter to identify the email in the callback method. You can register a handler for the SendCompleted event of the client to handle the result of the asynchronous operation, such as:

// Register a handler for the SendCompleted event
client.SendCompleted += new SendCompletedEventHandler(SendCompletedCallback);

// Define the callback method
private static void SendCompletedCallback(object sender, AsyncCompletedEventArgs e)
{
    // Get the user token from the event arguments
    string token = (string) e.UserState;

    // Check if the operation was canceled
    if (e.Cancelled)
    {
        Console.WriteLine("[{0}] Send canceled.", token);
    }

    // Check if there was an error
    if (e.Error != null)
    {
        Console.WriteLine("[{0}] {1}", token, e.Error.ToString());
    }
    else
    {
        Console.WriteLine("Message sent.");
    }
}

For more information, you can refer to the official documentation: https://docs.microsoft.com/en-us/dotnet/api/system.net.mail.smtpclient.sendasync?view=net-6.0
