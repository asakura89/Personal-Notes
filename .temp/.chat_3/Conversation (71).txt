using System;
using System.IO;

class Program
{
    static void Main(string[] args)
    {
        // Example of a fullpath string
        string fullpath = @"C:\Users\Documents\Report.pdf";
        
        // Use the Path.GetFileName method to get the filename with extension
        string filename = Path.GetFileName(fullpath);
        
        // Print the result
        Console.WriteLine("The filename with extension is: " + filename);
    }
}
